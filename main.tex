\begin{abstract}
We study the expressive power of symmetric two-counter machines, namely
two-counter machines whose transitions
are invariant under swapping of counters.
\end{abstract}


\newpage

\section{The model}

A 2-\vass $\V$ is \emph{symmetric} if whenever it has a transition
$q \xstep{(x_1,x_2)} q'$
then it also has a the transposed transition 
$q \xstep{(x_2,x_1)} q'$.
We consider \emph{paths}, which are sequences of transitions where target state 
of every transition
matches the source targe state of the next one.
%
A \emph{run} is a path plus a source configuration whose
state matches with the first transition, such that
all transitions are fireable.
A \emph{cycle} is a path where the source state of the 
first matches the target state of the last one.
By $c; c'$ we denote the concatenation of paths $c, c'$, and  
by $c^k$ we denote the $k$-fold concatenation of a cycle $c$.
Firing of a single transition we call a \emph{step} and
write $q(n_1, n_2) \step q'(n_1+x_1, n_2+x_2)$.

\medskip

We denote by $\mysize \V$ the size of $\V$, in unary representation,
and by $\norm \V$ the max-norm of $\V$, namely the maximal absolute value of
all numbers appearing in transitions.
Clearly, $\mysize \V \geq \norm \V$.

\section{Target pummping}

\begin{lemma} [Target pumping] \label{lem:targetpump}
Given a symmetric 2-\vass $\V$ one can compute numbers $C, D \in \N$ 
such that for every pair of control states $q, q'$ of $\V$ and  
$a, a'\in\N$ such that $a, a'-a > C$, 
\[
q(a,0)\step^* q'(0,a') \quad \implies  \quad  
q(a,0) \step^* q'(0, a'+D).
\]
Moreover, $C$ is polynomial in $\mysize \V$ and $D$ is exponential in $\mysize \V$.
\end{lemma}
%
\begin{proof}
Consider a run $\sigma : q(a,0)\step^* q'(0,a')$ for some $a, a'\in \N$.
We modify the run to lift the target configuration to $q'(0,a'+\gamma D_\sigma)$ for any
$\gamma > 0$, for some particular $D_\sigma$ depending both on $\V$ and on $\sigma$, assuming
that both $a$ and the difference $a'-a$ are sufficiently large (namely larger
than some $C$, depending now only on $\V$).
As there will be only finitely many different numbers $D_\sigma$,
we argue then that taking the least common multiple of all $D_\sigma$ yields a
number $D$ satisfying the claim of the lemma.

By the LPS decomposition \cite{Blondin15}, we may assume \mywlog that 
$\sigma$ consists of a polynomial number of transitions $t_0, \ldots, t_\ell$
and cycles $c_1, \ldots, c_\ell$, each cycle of
polynomial length  (with respect to the size $\mysize \V$ of
unary representation of $\V$):
\[
\sigma \ \in \ s_0 \ c_1^* \ s_1 \ c_2^* \ \ldots \ c_\ell^* \ s_\ell,
\]
where the length of each cycle $c_i$ is at most $e$,
and both $\ell$ and $e$ are bounded by a polynomial of $\mysize \V$.
Let $C_0$ be the minimal value such that every cycle of length at most $e$ 
is fireable at $(C_0, C_0)$.

We now classify the cycles as follows: call a cycle $c_j$ \emph{major}
if its number of iterations $n_j$ is larger than $2 C_0$, and \emph{minor} otherwise.
Moreover, call a cycle
$c_j$ \emph{sum-increasing} if its effect $(x_j, y_j)$ satisfies $x_j + y_j > 0$.

As we're assuming that $a' - a > C$, note that the sum of counters
must increase by more by more than $C$ along $\sigma$. Thus, by setting an appropriately
large $C$, we can ensure that at least one \emph{major sum-increasing cycle}
will be present in the decomposition of $\sigma$.

In particular, the total effect of the one-off transitions $s_0, \ldots, s_\ell$ is bounded,
as each of them may increase the sum of the counters by at most $\norm \V$.
Likewise, each minor cycle may only increase the sum of the counters by at most
$2 C_0 e \norm \V$. Thus, it suffices to set
\[
    C = (2 C_0 e_\mathrm{max} + 1)(\ell_\mathrm{max} + 1)\norm\V,
\]
where $e_\mathrm{max}$ and $\ell_\mathrm{max}$ are the (polynomial in $\mysize\V$) bounds
on $e$ and $\ell$ respectively.

\section{Proof strategy}

\sla{This is just a sketch of the proof structure, to be completed, debugged, etc}
As above, we use $\step$ to denote step relation of a 2-\vass, namely without zero tests.

For convenience we assume that instead of zero testing transitions which change state
when a given counter is 0, we have "zero-testing states" which can be visited only
if a given counter is 0. Transformation to this form is easy: replace every
zero testing transition by two, with intermediate zero-testing state.

\para{1-dimensional notation}
Likewise, we prove the three remaining variants of \Cref{lem:targetpump}:
for every pair of control states $q, q'$ of $\V$ and  
$a, a'\in\N$ such that $a, a'-a > C$, 
\begin{align*}
q(0,a)\step^* q'(0,a') \quad \implies  \quad  q(0,a) \step^* q'(0, a'+D)\\
q(0,a)\step^* q'(a',0) \quad \implies  \quad  q(0,a) \step^* q'(a'+D, 0)\\
q(a,0)\step^* q'(a',0) \quad \implies  \quad  q(a,0) \step^* q'(a'+D,0)
\end{align*}
By the above introduced "zero-testing states" we can
merge the four cases into one, knowing that  control state determines which
counter is actually zero tested. %, and that a run zero tests every counter whenever it has value 0. 
% To this aim double the control states, that is replace every $q\in Q$
% by $q_1, q_2$, each of them involved into the same increment transitions as $q$, but $q_1$ only 
% capable to do those zero-testing transitions $q$ that test counter 1, and likewiese $q_2$ only 
% capable to do those zero-testing transitions $q$ that test counter 2.
So prepared, we can simply write
\begin{align} \label{eq:basictran}
q(a) \step^* q'(a')
\end{align}
without specifying which of the four cases is happening
\begin{align*}
q(a,0)\step^* q'(0,a')\\
q(0,a)\step^* q'(0,a')\\
q(0,a)\step^* q'(a',0)\\
q(a,0)\step^* q'(a',0)
\end{align*}
since the control states $q,q'$ uniquely determine these two bits of information.
Note that the newly introduced zero tests are not symmetric, but we do not assume that. 

\Mywlog we assume that both the source and target vector have at least one 0.

\para{Run structure}

We split the relation \eqref{eq:basictran} into 3 disjoint subrelations, depending on the value of
$a, a'$ and $a'-a$. We write
\begin{align} \label{eq:tran0-+}
\begin{aligned}
& q(a) \step_0 q'(a') && \text{ if } a\leq C \vee a'\leq C \vee |a'-a|\leq C \\
& q(a) \step_- q'(a') && \text{ if none of above holds but } a > a'+C \\
& q(a) \step_+ q'(a') && \text{ if none of above holds but } a' > a+C
\end{aligned}
\end{align}

Each of the three relations \eqref{eq:tran0-+} is included in $(Q\times\N)^2$.
In particular, the four variants of \Cref{lem:targetpump} jointly imply that 
$\step_+$ is \emph{target-pumpable}, that is for every pair of control states $q, q'$ of $\V$, 
% and  $a, a'\in\N$ such that $a, a'-a > C$, 
\begin{align} \label{eq:tpump}
q(a)\step_+ q'(a') \quad \implies  \quad  q(a) \step_+ q'(a'+D).
\end{align}
The reverse version of the four variants, where we swap source and target and reverse all transitions, 
implies that $\step_-$ is \emph{source-pumpable}, that is 
\begin{align} \label{eq:spump}
q(a)\step_- q'(a') \quad \implies  \quad  q(a+D) \step_- q'(a').
\end{align}

The idea is to split every run into segments, at positions where some of counters is zero tested.
The segment between two such consecutive positions is described by one of the three relations
$\step_0$, $\step_-$ and $\step_+$.
Now the rechability relation is $\big(\step_0 \ \cup \ \step_- \ \cup \ \step_+ \big)^*$, an arbitrary composition
of all the three relations arbitrarily many times, where $R^*$ stands for composing the relation $R$
arbitrarily
many times.
Note that the composition assures that control states match.
We prefer to see the reachability relation as described by the expression
\begin{align} \label{eq:3phases}
\big(\step_{0}^* \ \circ \ E \ \circ \ \step_+)^* \ \circ \ \step_0^* \ \circ \ E
\end{align}
where 
\[
 E  \ = \ \big(\step_- \ \circ \ \step_0^*\big)^*.
\]
Thus we split every run into three phases: the first one ends with the last appearance of $\step_+$,
the middle one consisting just of $\step_0^*$, and the last one starting with $\step_-$ and
iterating arbitratily many times $\step_-$ and $\step_0$.
Each of the phases may be empty.


\para{Computing pumpable sets}

In the sequel we compute sets of configurations $X\subseteq Q\times\N$ which are \emph{pumpable},
namely $q(a)\in X \implies q(a+D)\in X$. We represent these sets in vector form as functions
$X:Q\to \powset{\N}$, where $X(q)$ is pumpable for every $q\in Q$, namely $a\in X(q)\implies a+D\in X(q)$.
Termination of computation will be guaranteed since the chains of such sets $X$ will be increasing, and

\begin{lemma} \label{lem:stab}
    Every increasing chain of pumpable subset of $\N$ stabilizes.
\end{lemma}

The two basic building blocks:

\begin{lemma} \label{lem:0-+}
The relations $\step_0, \ \step_-, \ \step_+$ are effectively semi-linear (and thus computable).
\end{lemma}
\begin{proof}
    They are sections of the reachability relation of a 2-\vass which is effectively semi-linear \cite{Blondin15}.
\end{proof}

\begin{lemma} \label{lem:0star}
The relation $\step_0^*$ is effectively semi-linear (and thus computable).
\end{lemma}
\begin{proof}
    Computable by a 1-counter automaton, a special case of a pushdown automaton.
\end{proof}

Notationally, we allows ourselves to compose a relation $R\subseteq (Q\times\N)^2$ 
with a set $X\subseteq Q\times\N$ to denote the set inverse image $R\circ X\subseteq Q\times\N$, 
or the image $X\circ R \subseteq Q\times\N$.
% Let $I_s = \{(s,s)\}$ be the relation containg exactly one pair, where $s$ is the source configuration.
% Likewise, let $I_t = \{(t,t)\}$.
We will be able to decide if $s\step^* t$, given source and target configurations $s,t\in Q\times\N$,
as long as we prove the two following fact: one can (backward) compute the inverse image
$E\circ \{t\}$, and (forward) compute the image 
$\{s\} \circ \big(\step_{0}^* \ \circ \ E \ \circ \ \step_+)^*$,
according to the partition into phases \eqref{eq:3phases}.

\begin{lemma} \label{lem:compback}
One can compute 
the (source-pumpable) set $\big(\step_- \ \circ \ \step_0^*\big)^* \ \circ \ \{t\}$.
\end{lemma}
\begin{proof}
The set $\big(\step_- \ \circ \ \step_0^*\big) \ \circ \ S$ is source-pumpable due to \eqref{eq:spump},
irrespectively of $S$, and computable due to \Cref{lem:0-+,lem:0star}, as long $S$ is semi-linear.
The claim follows by iterating the computation of the transformation
\[
S \quad \mapsto \quad \big(\step_- \ \circ \ \step_0^*\big) \ \circ \ S,
\]
relying on \Cref{lem:stab}.
\end{proof}

\begin{lemma} \label{lem:compfor}
One can compute 
the (target-pumpable) set 
$\{s\} \circ \big(\step_{0}^* \ \circ \ E \ \circ \ \step_+)^*$.
\end{lemma}
\begin{proof}
Again, The set $S \circ \step_{0}^* \ \circ \ E \ \circ \ \step_+$ is
target-pumpable due to \eqref{eq:tpump},
irrespectively of $S$.
Computability is more involved, but relates again on iteration of the transformation
\[
S \quad \mapsto \quad f(S) \ = \ S \ \circ \ \step_{0}^* \ \circ \ E \ \circ \ \step_+,
\]
relying on \Cref{lem:stab}.
To compute $f(S)$, one first establishes which remainders modulo $D$ appear in this set,
by applyting the transformation of \Cref{lem:compback} to the set $\step_+ \ \circ \ M_k$,
where $M_k$ contains all numbers equal to $k$ modulo $D$, and checking intersection with 
$S\ \circ \ \step_0$. 
Then, by exhaustive search one finds the smallest number in $f(S)$, 
for each remainder modulo $D$ separately.
\end{proof}
