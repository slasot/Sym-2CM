\begin{abstract}
We study the expressive power of symmetric two-counter machines, namely
two-counter machines whose transitions
are invariant under swapping of counters.
\end{abstract}


\newpage

\section{The model}

A 2-\vass $\V$ is \emph{symmetric} if whenever it has a transition
$q \xstep{(x_1,x_2)} q'$
then it also has a the transposed transition 
$q \xstep{(x_2,x_1)} q'$.
We consider \emph{paths}, which are sequences of transitions where target state 
of every transition
matches the source targe state of the next one.
%
A \emph{run} is a path plus a source configuration whose
state matches with the first transition, such that
all transitions are fireable.
A \emph{cycle} is a path where the source state of the 
first matches the target state of the last one.
By $c; c'$ we denote the concatenation of paths $c, c'$, and  
by $c^k$ we denote the $k$-fold concatenation of a cycle $c$.
Firing of a single transition we call a \emph{step} and
write $q(n_1, n_2) \step q'(n_1+x_1, n_2+x_2)$.

\medskip

We denote by $\mysize \V$ the size of $\V$, in unary representation,
and by $\norm \V$ the max-norm of $\V$, namely the maximal absolute value of
all numbers appearing in transitions.
Clearly, $\mysize \V \geq \norm \V$.

\section{Target pummping}

\begin{lemma} [Target pumping]
Given a symmetric 2-\vass $\V$ one can compute numbers $C, D \in \N$ 
such that for every pair of control states $q, q'$ of $\V$ and  
$a, a'\in\N$ such that $a, a'-a > C$, 
\[
q(a,0)\step^* q'(0,a') \quad \implies  \quad  
q(a,0) \step^* q'(0, a'+D).
\]
Moreover, $C$ is polynomial in $\mysize \V$ and $D$ is exponential in $\mysize \V$.
\end{lemma}
%
\begin{proof}
Consider a run $\sigma : q(a,0)\step^* q'(0,a')$ for some $a, a'\in \N$.
We modify the run to lift the target configuration to $q'(0,a'+D_\sigma)$ for some 
$D_\sigma$ depending both on $\V$ and on $\sigma$, assuming
that both $a$ and the difference $a'-a$ are sufficiently large (namely larger
than some $C$, depending now only on $\V$).
As there will be only finitely many different numbers $D_\sigma$,
we argue then that taking the least common multiple of all $D_\sigma$ yields a
number $D$ satisfying the claim of the lemma.

By the LPS decomposition \cite{Blondin15}, we may assume \mywlog that 
$\sigma$ consists of a polynomial number of transitions $t_0, \ldots, t_\ell$
and cycles $c_1, \ldots, c_\ell$, each cycle of
polynomial length  (with respect to the size $\mysize \V$ of
unary representation of $\V$):
\[
\sigma \ \in \ s_0 \ c_1^* \ s_1 \ c_2^* \ \ldots \ c_\ell^* \ s_\ell,
\]
where the length of each cycle $c_i$ is at most $e$,
and both $\ell$ and $e$ are bounded by a polynomial of $\mysize \V$.
Let $C_0$ be the minimal value such that every cycle of length at most $e$ 
is fireable
at $(C_0, C_0)$.
Assume that $a$ and $a'-a$ are both sufficiently large 
(i.e. larger than $C$, to be defined later).
Thus the sum of counters increases by sufficiently large value 
(larger than $C$) along $\sigma$.

The total effect of transitions $t_0, \ldots, t_\ell$ is negligible as they
may increase the sum of counters by at most $(\ell+1)\norm \V$.
As there are at only $\ell$ cycles $c_1, \ldots, c_\ell$, iteration of some of them,
say $c_j$, necessarily increases the sum of counters by a sufficiently large value.
    \mei{We could use some more precision in describing how the increase of the counter sum
    gets distributed across the cycles, and how the respective uses of ``\emph{sufficiently
    large}'' relate to each other.}
This means that $c_j$ itself increases the sum of counters, and that the number $n_j$ of
iterations of the cycle $c_j$ is sufficiently large.
Call every cycle $c_j$ satisfying the two above conditions \emph{sum-increasing}.
The effect of every sum-increasing cycle can be:
vertical $(0,+)$, horizontal $(+,0)$, or one of $(+,-)$, $(+,+)$, or $(-,+)$.

\para{Case 1: Some sum-increasing cycle $c_j$ has vertical $(0,+)$ effect}
One extra iteration of this cycle yields the claim with $D_\sigma$ equal to its
vertical effect.

\para{Case 2: Some sum-increasing cycle $c_j$ has $(-,+)$ or $(+,-)$ effect}
Assume $(-,+)$ effect, the other case is similar.
\sla{check}

For our argument it is enough if the $n_j$ is large enough to guarantee that 
in the middle
of $c_j^{n_j}$, namely after the half of $n_j$ of iterations of $c_j$, the 
transposed cycle $\overline{c_j}$ is also enabled.
(the transposed cycle $\overline{c_j}$ is obtained by transposing all transitions 
along $c_j$).
We observe the key claim, relying on the fact that  
the effect of $c_j$ nonzero on both coordinates 
(it seems that the fact is quite general and 
does not need the assumption that $c_j$ increases 
the sum of counters):
%
\begin{claim}  \label{claim:fireable}
If $n_j > 2C_0$, in the middle of $c_j^{n_j}$ the transposed 
cycle $\overline{c_j}$ is fireable
(irrspectively of the source configuration). 
\end{claim}
%
\begin{proof}
The $n_j+1$ source-target configurations along the run $c_j^{n_j}$
form a linear progression.
As the effect of $c_j$ is nonzero on both coordinates,
the middle source-target point is above $(C_0, C_0)$.
\end{proof}
%

In consequence of \Cref{claim:fireable} the cycle $\overline{c_j}$ is fireable
in the middle of $c_j^{n_j}$, 
and hence the composition $\overline{c_j}; c_j$ is fireable too.
\mei{How do we obtain that the composition is fireable? Do we need to raise $C_0$ for this?}
Let $(-m_1, m_2)$ denote the effect of $c_j$. 
Thus $m_2 > m_1$, and
the effect of $\overline{c_j}; c_j$ is $(m_2-m_1,m_2-m_1)$.
% Let $k$ be the least common multiplicity of $m_2 - m_1$ and $m_1$.  
Let $k:=m_1(m_2-m_1)$.
Thus iterating $m_1$ times the cycle $\overline{c_j}; c_j$
we get a cycle with effect $(k,k)$, and then iterating
$m_2-m_1$ times the cycle $\overline {c_j}$ we get the vertical effect.
Indeed, the effect of
\[
\big(\overline{c_j}; c_j\big)^{m_1}; c_j^{m_2-m_1}
\]
is $(0, (m_1+m_2)(m_2-m_1))$, and this cycle is still fireable in the middle of
$c_j^{n_j}$.
We thus get the claim with $D_\sigma = (m_1+m_2)(m_2-m_1)$.

\para{Case 3: All sum-increasing cycles have $(+,0)$ or $(+,+)$ effect}
Let us restrict our attention to those cycles $c_j$, whose effect $(m_1,m_2)$ satisfies $m_1\neq 0$.
Depending on whether $m_1>0$ or $m_1<0$, we call such cycles 
horizontally positive, or horizontally negative.  
By the above assumption, every sum-increasing cycle is horizontally positive;
note however that horizontally negative cycles that aren't sum-increasing can
still be present.
We define \emph{steepness} of $c_j$ as $m_2/m_1$.

\begin{claim} \label{claim:steep}
There are some $i<j$ such that $c_i$ is sum-increasing, $c_j$ is 
horizontally negative, and $c_i$ is steeper than $c_j$.
\end{claim}
%
\begin{proof}
Supposing there is no such $i<j$, we have that whenever $c_i$ is sum-increasing,
every horizontally-negative cycle $c_j$, with $j>i$, is at least as steep as $c_i$.
This contradicts $a'-a>C$.
\sla{check and fill details}
\end{proof}

By \Cref{claim:steep}, for some positive integers $I, J$, the sum of effects of
$c_i^I$ and $c_j^J$ is vertical $(0,+)$.
Extending $c_i^{n_i}$ with $c_i^I$ and $c_j^{n_j}$ with $c_j^J$ yields
a run ending in $q'(0,a'+D_\sigma)$, where $D_\sigma = ,,,$.
\sla{complete}

\medskip

The norm of the effect of every cycle of length at most $e$ is bounded
by $e\cdot\norm{\V}$, and hence
All values $D_\sigma$ are bounded by $2\cdot e^2\cdot \norm{\V}^2$, and hence
polynomially in $\mysize \V$.
Taking as $D$ the least common multiplicity of all $D_\sigma$ is enough for the
claim of the lemmma.

The constant $C$ must be large enough so that, even after forgetting of
the total effects of all transitions $t_0, \ldots, t_\ell$, the iterations of
cycles $c_1, \ldots, c_\ell$ increase the sum by the amount large enough so that
one of sum-incrasing cycles is iterated at least $2 C_0$ times (are reqiured
by \Cref{claim:fireable}).
As $C_0$ is at most $e\cdot \norm \V$ and $\ell$ is polynomial in $\mysize \V$,
a polynomial value of $C$, \wrt $\mysize \V$, 
is sufficient to satisfy the above requirement.
\sla{complete the calculations}
\end{proof}
